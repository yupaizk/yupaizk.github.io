function saveProfile(){
  const gBtn=document.querySelector('#pm-gender .pm-pill.active');
  const sBtn=document.querySelector('#pm-seek .pm-pill.active');
  const age=parseInt(document.getElementById('pm-age').value);
  if(!gBtn){alert('請選擇你的性別 😊');return;}
  if(!age||age<18||age>60){alert('請輸入正確的年齡（18–60歲）😊');return;}
  if(!sBtn){alert('請選擇你想配對的對象 😊');return;}
  userProfile.gender=gBtn.dataset.val;
  userProfile.age=age;
  userProfile.seek=sBtn.dataset.val;
  // sync to inline bar
  syncToBar();
  closeProfile();
  openQuiz();
}
function syncToBar(){
  const {gender,age,seek}=userProfile;
  if(!gender)return;
  const ab=age<25?'20–24歲':age<30?'25–29歲':age<35?'30–34歲':'35歲以上';
  document.getElementById('profileSummary').innerHTML=
    `✅ 已設定：${gender}・${age}歲（${ab}）・尋找 ${seek}`;
  // sync pills in bar
  document.querySelectorAll('#genderPills .pf-pill').forEach(b=>{b.classList.toggle('active',b.textContent.includes(gender));});
  document.querySelectorAll('#seekPills .pf-pill').forEach(b=>{b.classList.toggle('active',b.textContent.includes(seek));});
  if(document.getElementById('ageInput'))document.getElementById('ageInput').value=age;
}

function renderQ(){
  const q=questions[currentQ];
  document.getElementById('qsNum').textContent=`第 ${currentQ+1} 題 / 共 ${questions.length} 題`;
  document.getElementById('qBar').style.width=`${((currentQ+1)/questions.length)*100}%`;
  document.getElementById('qText').textContent=q.q;
  const el=document.getElementById('qOpts');el.innerHTML='';
  q.opts.forEach((o,i)=>{
    const b=document.createElement('button');b.className='quiz-opt';b.textContent=o;
    b.onclick=()=>answerQ(i);el.appendChild(b);
  });
}
function answerQ(i){
  quizAnswers.push(i);currentQ++;
  if(currentQ<questions.length){renderQ();}else{showQuizResult();}
}
